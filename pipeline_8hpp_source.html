44  double performance()
45  {
46  // actual transformation instances are weighted much higher
47  // so that we not only get a large number of keypoints
48  return 0.01 * (sourceDownsampledPoints / MAX_POINTS)
49  + 0.01 * (targetDownsampledPoints / MAX_POINTS)
50  + 0.03 * (sourceKeypointsExtracted / MAX_POINTS)
51  + 0.03 * (targetKeypointsExtracted / MAX_POINTS)

182  _keypointExtractor->extract(target, target_keypoints);

187  target_keypoints = PclPointCloud::Ptr(target.cloud());

191  stats.targetKeypointsExtracted = target_keypoints->size();

208  typename PclDescriptorCloud::Ptr source_features(new PclDescriptorCloud);
209  typename PclDescriptorCloud::Ptr target_features(new PclDescriptorCloud);

213  _featureDescriptor->describe(target, target_keypoints, target_keypoint_normals, target_features);

215 
216  // matching
217  Logger::log("Feature matching...");
218  pcl::CorrespondencesPtr correspondences(new pcl::Correspondences);
219  _featureMatcher->match(source_features, target_features, correspondences);
220 
221  for (size_t j = 0; j < std::max(correspondences->size(), 500); ++j) // limit the number of displayed corrs.
222  {
223  PointT &model_point = source_keypoints->at((*correspondences)[j].index_query);
224  PointT &scene_point = target_keypoints->at((*correspondences)[j].index_match);

227  Visualizer::visualize(model_point, scene_point, Color::random());

229 
230  stats.averageCorrespondenceDistance = FeatureMatcher<DescriptorT>::averageDistance(correspondences);
231  stats.correspondencesFound = correspondences->size();
232

235  std::vector<Eigen::Matrix4f> transformations;
236  bool tes = _transformationEstimator->estimate(
237  source, target,

239  source_features, target_features,

241  Logger::debug(boost::format("Transformation estimation successfull? %1%") % tes);
