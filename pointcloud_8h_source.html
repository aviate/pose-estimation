205  PclNormalCloud::Ptr normals(const PclPointCloud::Ptr &targetPoints)
206  {

208  PclNormalCloud::Ptr targetNormals(new PclNormalCloud);
209 
210  // accelerate search for equal points using an Kd tree
211  typename pcl::KdTreeFLANN<PointT>::Ptr kdtree(new pcl::KdTreeFLANN<PointT>);
212  kdtree->setInputCloud(_cloud);
213  size_t idx = 0;
214  std::vector<int> pointIdxNKNSearch(1);
215  std::vector<float> pointNKNSquaredDistance(1);
216  for (PointT &targetPoint : targetPoints->points)
217  {
218  if (kdtree->nearestKSearchT(targetPoint, 1, pointIdxNKNSearch, pointNKNSquaredDistance) < 0)
219  {
220  Logger::warning(boost::format("Normal estimation failed for target point %i. " \
221  "No matching point could be found in the point cloud.") % idx);
222  continue;
223  }
224 
225  targetNormals->push_back((*ns)[pointIdxNKNSearch[0]]);
226

252  void _computeResolution()
253  {
254  _resolution = 0.0;
255  int n_points = 0;
256  int nres;
257  std::vector<int> indices(2);
258  std::vector<float> sqr_distances(2);
259  pcl::search::KdTree<PointT> tree;
260  tree.setInputCloud(_cloud);
261 
262  for (size_t i = 0; i < _cloud->size(); ++i)
263  {
264  if (!pcl_isfinite((*_cloud)[i].x))
265  continue;
266 
267  // Considering the second neighbor since the first is the point itself.
268  nres = tree.nearestKSearch(i, 2, indices, sqr_distances);
269  if (nres == 2)
270  {
271  _resolution += sqrt(sqr_distances[1]);
272  ++n_points;
273  }
274  }
275  if (n_points != 0)
276  {
277  _resolution /= n_points;
278  }
279  else
280  {
281  Logger::warning("Point cloud resolution is 0. Setting it to 1.");
282  _resolution = 1.0;
283  }
284  }
