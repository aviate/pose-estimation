135  typename std::shared_ptr<FeatureDescriptor<PointT, DescriptorT> > &descriptor)
136  {
137  Pipeline<DescriptorT, PointT> pipeline;

140  switch (downsampler.value<Enum>().value)

142  case 1:
143  pipeline.downsampler() = std::make_shared<VoxelGridDownsampler<PointT> >();
144  break;
145  default:
146  pipeline.downsampler() = std::make_shared<UniformDownsampler<PointT> >();
147  }
148 
149  switch (featureMatcher.value<Enum>().value)
150  {
151  default:
152  pipeline.featureMatcher() = std::make_shared<KdTreeFeatureMatcher<DescriptorT> >();
153  }
154 
155  switch (keypointExtractor.value<Enum>().value)
156  {
157  case 1:
158  pipeline.keypointExtractor() = std::make_shared<ISSKeypointExtractor<PointT> >();
159  break;
160  default:
161  pipeline.keypointExtractor() = std::make_shared<UniformKeypointExtractor<PointT> >();
162  }
163 
164  switch (transformationEstimator.value<Enum>().value)
165  {
166  case 1:
167  pipeline.transformationEstimator() = std::make_shared<HoughVoting<PointT, DescriptorT> >();
168  break;
169  case 2:
170  pipeline.transformationEstimator() = std::make_shared<RANSACTransformationEstimator<PointT, DescriptorT> >();
171  break;
172  case 3:
173  pipeline.transformationEstimator() = std::make_shared<SVDTransformationEstimator<PointT, DescriptorT> >();
174  break;
175  default:
176  pipeline.transformationEstimator() = std::make_shared<GeometricConsistency<PointT, DescriptorT> >();
177  }
